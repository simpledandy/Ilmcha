// Helper function to get start and end points from path data
const getPathPoints = (pathData: string) => {
  const commands = pathData.match(/[MLHVCSQTAZ][^MLHVCSQTAZ]*/g) || [];
  const points: { x: number; y: number }[] = [];
  
  let currentX = 0;
  let currentY = 0;
  
  commands.forEach((command) => {
    const type = command[0];
    const coords = command.slice(1).trim().split(/[\s,]+/).map(Number);
    
    switch (type) {
      case 'M': // Move to
        currentX = coords[0];
        currentY = coords[1];
        points.push({ x: currentX, y: currentY });
        break;
      case 'L': // Line to
        currentX = coords[0];
        currentY = coords[1];
        points.push({ x: currentX, y: currentY });
        break;
      case 'H': // Horizontal line
        currentX = coords[0];
        points.push({ x: currentX, y: currentY });
        break;
      case 'V': // Vertical line
        currentY = coords[0];
        points.push({ x: currentX, y: currentY });
        break;
      case 'Q': // Quadratic curve - add control point and end point
        if (coords.length >= 4) {
          const cpX = coords[0];
          const cpY = coords[1];
          currentX = coords[2];
          currentY = coords[3];
          points.push({ x: cpX, y: cpY }); // Control point
          points.push({ x: currentX, y: currentY }); // End point
        }
        break;
      case 'C': // Cubic curve - add control points and end point
        if (coords.length >= 6) {
          const cp1X = coords[0];
          const cp1Y = coords[1];
          const cp2X = coords[2];
          const cp2Y = coords[3];
          currentX = coords[4];
          currentY = coords[5];
          points.push({ x: cp1X, y: cp1Y }); // First control point
          points.push({ x: cp2X, y: cp2Y }); // Second control point
          points.push({ x: currentX, y: currentY }); // End point
        }
        break;
      case 'S': // Smooth cubic curve
        if (coords.length >= 4) {
          const cp2X = coords[0];
          const cp2Y = coords[1];
          currentX = coords[2];
          currentY = coords[3];
          points.push({ x: cp2X, y: cp2Y }); // Second control point
          points.push({ x: currentX, y: currentY }); // End point
        }
        break;
      case 'T': // Smooth quadratic curve
        if (coords.length >= 2) {
          currentX = coords[0];
          currentY = coords[1];
          points.push({ x: currentX, y: currentY });
        }
        break;
      case 'A': // Arc - add start and end points
        if (coords.length >= 7) {
          currentX = coords[5];
          currentY = coords[6];
          points.push({ x: currentX, y: currentY });
        }
        break;
      case 'Z': // Close path - no new points, just close to start
        break;
    }
  });
  
  return {
    startPoint: points[0] || { x: 0, y: 0 },
    endPoint: points[points.length - 1] || { x: 0, y: 0 }
  };
};
 
// Generate tracing data for numbers
const generateNumberData = (): TracingItem[] => {
  return Object.entries(numberPaths).map(([number, pathData]) => {
    const { startPoint, endPoint } = getPathPoints(pathData);
    return {
      id: `number-${number}`,
      pathData,
      startPoint,
      endPoint,
      difficulty: number === '1' || number === '7' ? 'easy' : number === '8' ? 'hard' : 'medium',
      category: 'number',
      language: 'en',
      value: number
    };
  });
};
 
// Generate tracing data for English letters
const generateEnglishLetterData = (): TracingItem[] => {
  return Object.entries(englishLetterPaths).map(([letter, pathData]) => {
    const { startPoint, endPoint } = getPathPoints(pathData);
    return {
      id: `letter-en-${letter}`,
      pathData,
      startPoint,
      endPoint,
      difficulty: ['I', 'L', 'T'].includes(letter) ? 'easy' : ['B', 'R', 'S'].includes(letter) ? 'hard' : 'medium',
      category: 'letter',
      language: 'en',
      value: letter
    };
  });
};
 
// Generate tracing data for Uzbek letters
const generateUzbekLetterData = (): TracingItem[] => {

// Export all tracing data
export const tracingData: TracingItem[] = [
  ...generateNumberData(),
  ...generateEnglishLetterData(),
  ...generateUzbekLetterData()
];
 
// Helper functions to get specific data
export const getTracingDataByCategory = (category: 'number' | 'letter', language: 'en' | 'uz' = 'en') => {
  return tracingData.filter(item => item.category === category && item.language === language);
};
 
export const getTracingDataByDifficulty = (difficulty: 'easy' | 'medium' | 'hard', category?: 'number' | 'letter') => {
  return tracingData.filter(item => 
    item.difficulty === difficulty && 
    (category ? item.category === category : true)
  );
};
 
export const getTracingItem = (id: string): TracingItem | undefined => {
  return tracingData.find(item => item.id === id);
};
