import React, { useState, useEffect } from 'react';
import { View, Text, StyleSheet, Dimensions } from 'react-native';
import { PanGestureHandler, State } from 'react-native-gesture-handler';
import Animated, {
  useSharedValue,
  useAnimatedGestureHandler,
  useAnimatedStyle,
  runOnJS,
  interpolate,
  Extrapolate,
} from 'react-native-reanimated';
 
const { width: screenWidth, height: screenHeight } = Dimensions.get('window');
const canvasSize = Math.min(screenWidth * 0.8, screenHeight * 0.4);
 
interface TracingCanvasProps {
  pathData: string;
  strokeWidth?: number;
  strokeColor?: string;
  backgroundColor?: string;
  onComplete?: () => void;
  tolerance?: number;
  showGuide?: boolean;
  guideColor?: string;
  testID?: string;
}
 
export default function TracingCanvas({
  pathData,
  strokeWidth = 8,
  strokeColor = '#4CAF50',
  backgroundColor = '#F0F8FF',
  onComplete,
  tolerance = 20,
  showGuide = true,
  guideColor = '#E0E0E0',
  testID = 'tracing-canvas',
}: TracingCanvasProps) {
  const [error, setError] = useState<string | null>(null);
  const [progress, setProgress] = useState(0);
  const [isCompleted, setIsCompleted] = useState(false);
  const [tracedPoints, setTracedPoints] = useState<Set<number>>(new Set());
 
  const progressValue = useSharedValue(0);
  const isDrawing = useSharedValue(false);
 
  // Validate input props
  useEffect(() => {
    if (!pathData || typeof pathData !== 'string') {
      setError('Invalid path data provided');
      return;
    }
 
    if (tolerance < 0 || tolerance > 100) {
      setError('Tolerance must be between 0 and 100');
      return;
    }
 
    if (strokeWidth < 1 || strokeWidth > 50) {
      setError('Stroke width must be between 1 and 50');
      return;
    }
 
    setError(null);
  }, [pathData, tolerance, strokeWidth]);
 
  // Simple path parsing - just extract basic points
  const parsePathData = (path: string) => {
    try {
      if (!path || path.trim().length === 0) {
        console.log('Empty path data provided');
        return [];
      }
 
      console.log('Parsing path data:', path.substring(0, 50) + '...');
      const commands = path.match(/[MLHVCSQTAZ][^MLHVCSQTAZ]*/g) || [];
      
      if (commands.length === 0) {
        console.log('No valid commands found in path data');
        return [];
      }
 
      const points: { x: number; y: number }[] = [];
      
      let currentX = 0;
      let currentY = 0;
      
      commands.forEach((command, index) => {
        try {
          const type = command[0];
          const coords = command.slice(1).trim().split(/[\s,]+/).map(Number);
          
          // Validate coordinates
          if (coords.some(coord => isNaN(coord) || !isFinite(coord))) {
            console.warn(`Invalid coordinates in command ${index}:`, command);
            return;
          }
          
          switch (type) {
            case 'M': // Move to
              if (coords.length >= 2) {
                currentX = coords[0];
                currentY = coords[1];
                points.push({ x: currentX, y: currentY });
              }
              break;
            case 'L': // Line to
              if (coords.length >= 2) {
                currentX = coords[0];
                currentY = coords[1];
                points.push({ x: currentX, y: currentY });
              }
              break;
            case 'H': // Horizontal line
              if (coords.length >= 1) {
                currentX = coords[0];
                points.push({ x: currentX, y: currentY });
              }
              break;
            case 'V': // Vertical line
              if (coords.length >= 1) {
                currentY = coords[0];
                points.push({ x: currentX, y: currentY });
              }
              break;
            case 'Q': // Quadratic curve - just add end point
              if (coords.length >= 4) {
                currentX = coords[2];
                currentY = coords[3];
                points.push({ x: currentX, y: currentY });
              }
              break;
            case 'C': // Cubic curve - just add end point
              if (coords.length >= 6) {
                currentX = coords[4];
                currentY = coords[5];
                points.push({ x: currentX, y: currentY });
              }
              break;
            case 'S': // Smooth cubic curve - just add end point
              if (coords.length >= 4) {
                currentX = coords[2];
                currentY = coords[3];
                points.push({ x: currentX, y: currentY });
              }
              break;
            case 'T': // Smooth quadratic curve
              if (coords.length >= 2) {
                currentX = coords[0];
                currentY = coords[1];
                points.push({ x: currentX, y: currentY });
              }
              break;
            case 'A': // Arc - just add end point
              if (coords.length >= 7) {
                currentX = coords[5];
                currentY = coords[6];
                points.push({ x: currentX, y: currentY });
              }
              break;
            case 'Z': // Close path - no new points
              break;
            default:
              console.warn(`Unknown command type: ${type}`);
              break;
          }
        } catch (cmdError) {
          console.error('Error parsing command:', command, cmdError);
        }
      });
      
      console.log('Parsed points:', points.length);
      return points;
    } catch (parseError) {
      console.error('Error parsing path data:', parseError);
      setError('Failed to parse path data');
      return [];
    }
  };
 
  // Scale and center points to fit canvas
  const scaleAndCenterPoints = (points: { x: number; y: number }[]) => {
    try {
      if (points.length === 0) {
        console.log('No points to scale');
        return [];
      }
 
      // Find bounds
      const minX = Math.min(...points.map(p => p.x));
      const maxX = Math.max(...points.map(p => p.x));
      const minY = Math.min(...points.map(p => p.y));
      const maxY = Math.max(...points.map(p => p.y));
 
      const pathWidth = maxX - minX;
      const pathHeight = maxY - minY;
 
      // Handle zero-dimension paths
      if (pathWidth === 0 && pathHeight === 0) {
        console.log('Path has zero dimensions, using default size');
        return points.map(point => ({
          x: canvasSize / 2,
          y: canvasSize / 2,
        }));
      }
 
      console.log('Path bounds:', { minX, maxX, minY, maxY, pathWidth, pathHeight });
 
      // Calculate scale to fit in canvas with padding
      const padding = 40;
      const scaleX = (canvasSize - padding) / Math.max(pathWidth, 1);
      const scaleY = (canvasSize - padding) / Math.max(pathHeight, 1);
      const scale = Math.min(scaleX, scaleY, 5); // Limit max scale
 
      console.log('Scaled points:', points.length, 'Scale:', scale);
 
      // Scale and center points
      const centerX = (minX + maxX) / 2;
      const centerY = (minY + maxY) / 2;
      const canvasCenter = canvasSize / 2;
 
      return points.map(point => ({
        x: (point.x - centerX) * scale + canvasCenter,
        y: (point.y - centerY) * scale + canvasCenter,
      }));
    } catch (scaleError) {
      console.error('Error scaling points:', scaleError);
      setError('Failed to scale points');
      return [];
    }
  };
 
  const rawPathPoints = parsePathData(pathData);
  const pathPoints = scaleAndCenterPoints(rawPathPoints);
 
  // Generate simple tracing points
  const generateTracingPoints = () => {
    try {
      if (pathPoints.length === 0) {
        console.log('No path points to generate tracing points from');
        return [];
      }
 
      const tracingPoints: { x: number; y: number }[] = [];
      
      for (let i = 0; i < pathPoints.length - 1; i++) {
        const start = pathPoints[i];
        const end = pathPoints[i + 1];
        
        // Add start point
        tracingPoints.push(start);
        
        // Add intermediate points every 10 pixels
        const distance = Math.sqrt(Math.pow(end.x - start.x, 2) + Math.pow(end.y - start.y, 2));
        const steps = Math.max(1, Math.floor(distance / 10));
        
        for (let j = 1; j < steps; j++) {
          const ratio = j / steps;
          tracingPoints.push({
            x: start.x + (end.x - start.x) * ratio,
            y: start.y + (end.y - start.y) * ratio,
          });
        }
      }
      
      // Add the last point
      if (pathPoints.length > 0) {
        tracingPoints.push(pathPoints[pathPoints.length - 1]);
      }
      
      console.log('Generated tracing points:', tracingPoints.length);
      return tracingPoints;
    } catch (genError) {
      console.error('Error generating tracing points:', genError);
      setError('Failed to generate tracing points');
      return [];
    }
  };
 
  const tracingPoints = generateTracingPoints();
 
  // Simple proximity check
  const checkProximity = (x: number, y: number) => {
    try {
      if (tracingPoints.length === 0) {
        return { distance: Infinity, index: -1 };
      }
 
      let closestDistance = Infinity;
      let closestIndex = -1;
 
      tracingPoints.forEach((point, index) => {
        const distance = Math.sqrt(
          Math.pow(x - point.x, 2) + Math.pow(y - point.y, 2)
        );
        if (distance < closestDistance) {
          closestDistance = distance;
          closestIndex = index;
        }
      });
 
      return { distance: closestDistance, index: closestIndex };
    } catch (proximityError) {
      console.error('Error checking proximity:', proximityError);
      return { distance: Infinity, index: -1 };
    }
  };
 
  // Simple gesture handler
  const gestureHandler = useAnimatedGestureHandler({
    onStart: (event) => {
      try {
        console.log('Gesture started at:', event.x, event.y);
        isDrawing.value = true;
        
        if (tracingPoints.length === 0) {
          console.log('No tracing points available');
          return;
        }
        
        runOnJS(() => {
          const { distance } = checkProximity(event.x, event.y);
          if (distance <= tolerance) {
            progressValue.value = 0;
            setTracedPoints(new Set());
          }
        })();
      } catch (startError) {
        console.error('Error in gesture start:', startError);
        runOnJS(setError)('Gesture start error');
      }
    },
    onActive: (event) => {
      try {
        if (!isDrawing.value) return;
        
        if (tracingPoints.length === 0) {
          return;
        }
 
        runOnJS(() => {
          const { distance, index } = checkProximity(event.x, event.y);
          
          if (distance <= tolerance && index >= 0) {
            const newProgress = Math.max(progressValue.value, (index + 1) / tracingPoints.length);
            progressValue.value = newProgress;
            
            setTracedPoints((prev) => {
              const newSet = new Set(prev);
              newSet.add(index);
              return newSet;
            });
            
            if (newProgress >= 0.9 && !isCompleted) {
              setIsCompleted(true);
              if (onComplete) {
                onComplete();
              }
            }
            
            setProgress(newProgress * 100);
          }
        })();
      } catch (activeError) {
        console.error('Error in gesture active:', activeError);
        runOnJS(setError)('Gesture active error');
      }
    },
    onEnd: () => {
      try {
        console.log('Gesture ended');
        isDrawing.value = false;
      } catch (endError) {
        console.error('Error in gesture end:', endError);
        runOnJS(setError)('Gesture end error');
      }
    },
  });
 
  const animatedStyle = useAnimatedStyle(() => {
    return {
      opacity: interpolate(
        progressValue.value,
        [0, 1],
        [0.3, 1],
        Extrapolate.CLAMP
      ),
    };
  });
 
  const progressStyle = useAnimatedStyle(() => {
    return {
      width: `${progressValue.value * 100}%`,
    };
  });
 
  // Simple render function
  const renderTracingPath = () => {
    try {
      if (tracingPoints.length === 0) {
        return (
          <View style={styles.pathContainer}>
            <Text style={styles.errorText}>No path to trace</Text>
          </View>
        );
      }
 
      return (
        <View style={styles.pathContainer}>
          {tracingPoints.map((point, index) => (
            <View
              key={index}
              style={[
                styles.tracingDot,
                {
                  left: point.x - 4,
                  top: point.y - 4,
                  backgroundColor: tracedPoints.has(index) ? strokeColor : guideColor,
                  opacity: tracedPoints.has(index) ? 1 : 0.6,
                },
              ]}
            />
          ))}
        </View>
      );
    } catch (renderError) {
      console.error('Error rendering tracing path:', renderError);
      return (
        <View style={styles.pathContainer}>
          <Text style={styles.errorText}>Error rendering path</Text>
        </View>
      );
    }
  };
 
  // Show error if there is one
  if (error) {
    return (
      <View style={styles.container} testID={testID}>
        <View style={[styles.canvas, { backgroundColor }]}>
          <View style={styles.errorContainer}>
            <Text style={styles.errorText}>Error: {error}</Text>
            <Text style={styles.errorText}>Please try again</Text>
          </View>
        </View>
      </View>
    );
  }
 
  return (
    <View style={styles.container} testID={testID}>
      <View style={[styles.canvas, { backgroundColor }]}>
        <PanGestureHandler onGestureEvent={gestureHandler}>
          <Animated.View style={[styles.gestureArea, animatedStyle]}>
            {renderTracingPath()}
          </Animated.View>
        </PanGestureHandler>
      </View>
      
      {/* Progress bar */}
      <View style={styles.progressContainer}>
        <View style={styles.progressBar}>
          <Animated.View style={[styles.progressFill, progressStyle]} />
        </View>
        <Text style={styles.progressText}>{Math.round(progress)}%</Text>
      </View>
      
      {/* Status text */}
      <Text style={styles.statusText}>
        {isCompleted ? 'Completed! 🎉' : 'Trace the path'}
      </Text>
    </View>
  );
}

    alignItems: 'center',
